127 """
128 flatten(btdata, lidict) -> (newdata, newdict)
129
130 Produces "flattened" backtrace data. Individual instruction pointers
131 sometimes correspond to a multi-frame backtrace due to inlining; in
132 such cases, this function inserts fake instruction pointers for the
133 inlined calls, and returns a dictionary that is a 1-to-1 mapping
134 between instruction pointers and a single StackFrame.
135 """
136 function flatten(data::Vector{UInt}, lidict::Dict{UInt,Vector{StackFrame}})
137 # Makes fake instruction pointers, counting down from typemax(UInt)
138 newip = typemax(UInt)
139 taken = Set(keys(lidict)) # make sure we don't pick one that's already used
140 newdict = Dict{UInt,StackFrame}()
141 newmap = Dict{UInt,Vector{UInt}}()
142 for (ip, trace) in lidict
143 if length(trace) == 1
144 newdict[ip] = trace[1]
145 else
146 newm = UInt[]
147 for sf in trace
148 while newip âˆˆ taken && newip > 0
149 newip -= 1
150 end
151 newip == 0 && error("all possible instruction pointers used")
152 push!(newm, newip)
153 newdict[newip] = sf
154 newip -= 1
155 end
156 newmap[ip] = newm
157 end
158 end
159 newdata = UInt[]
160 for ip in data
161 if haskey(newmap, ip)
162 append!(newdata, newmap[ip])
163 else
164 push!(newdata, ip)
165 end
166 end
167 newdata, newdict
168 end

238 like [`clear`](:func:`clear`), can affect `data` unless you first make a copy. Note that the
239 values in `data` have meaning only on this machine in the current session, because it

258 # Counts the number of times each line appears, at any nesting level
259 function count_flat{T<:Unsigned}(data::Vector{T})
260 linecount = Dict{T,Int}()
261 toskip = btskip
262 for ip in data
263 if toskip > 0
264 toskip -= 1
265 continue
266 end
267 if ip == 0
268 toskip = btskip
269 continue
270 end
271 linecount[ip] = get(linecount, ip, 0)+1
272 end
273 iplist = Array{T}(0)
274 n = Array{Int}(0)
275 for (k,v) in linecount
276 push!(iplist, k)
277 push!(n, v)
278 end
279 return iplist, n
280 end

282 function parse_flat(iplist, n, lidict, C::Bool)
283 # Convert instruction pointers to names & line numbers
284 lilist = [lidict[ip] for ip in iplist]
285 # Keep only the interpretable ones
286 # The ones with no line number might appear multiple times in a single
287 # backtrace, giving the wrong impression about the total number of backtraces.
288 # Delete them too.
289 keep = !Bool[x == UNKNOWN || x.line == 0 || (x.from_c && !C) for x in lilist]
290 n = n[keep]
291 lilist = lilist[keep]
292 lilist, n
293 end

370 ## A tree representation
371 # Identify and counts repetitions of all unique backtraces
372 function tree_aggregate{T<:Unsigned}(data::Vector{T})
373 iz = find(data .== 0) # find the breaks between backtraces
374 treecount = Dict{Vector{T},Int}()
375 istart = 1+btskip
376 for iend in iz
377 tmp = data[iend-1:-1:istart]
378 treecount[tmp] = get(treecount, tmp, 0)+1
379 istart = iend+1+btskip
380 end
381 bt = Array{Vector{T}}(0)
382 counts = Array{Int}(0)
383 for (k,v) in treecount
384 if !isempty(k)
385 push!(bt, k)
386 push!(counts, v)
387 end
388 end
389 bt, counts
390 end

394 function tree_format(lilist::Vector{StackFrame}, counts::Vector{Int}, level::Int, cols::Integer)
395 nindent = min(cols>>1, level)
396 ndigcounts = ndigits(maximum(counts))
397 ndigline = maximum([tree_format_linewidth(x) for x in lilist])
398 ntext = cols-nindent-ndigcounts-ndigline-5
399 widthfile = floor(Integer,0.4ntext)
400 widthfunc = floor(Integer,0.6ntext)
401 strs = Array{String}(length(lilist))
402 showextra = false
403 if level > nindent
404 nextra = level-nindent
405 nindent -= ndigits(nextra)+2
406 showextra = true
407 end
408 for i = 1:length(lilist)
409 li = lilist[i]
410 if li != UNKNOWN
411 base = " "^nindent
412 if showextra
413 base = string(base, "+", nextra, " ")
414 end
415 if li.line == li.pointer
416 strs[i] = string(base,
417 rpad(string(counts[i]), ndigcounts, " "),
418 " ",
419 "unknown function (pointer: 0x",
420 hex(li.pointer,2*sizeof(Ptr{Void})),
421 ")")
422 else
423 fname = string(li.func)
424 if !li.from_c && !isnull(li.linfo)
425 fname = sprint(show_spec_linfo, li)
426 end
427 strs[i] = string(base,
428 rpad(string(counts[i]), ndigcounts, " "),
429 " ",
430 rtruncto(string(li.file), widthfile),
431 ":",
432 li.line == -1 ? "?" : string(li.line),
433 "; ",
434 ltruncto(fname, widthfunc))
435 end
436 else
437 strs[i] = ""
438 end
439 end
440 strs
441 end

443 # Print a "branch" starting at a particular level. This gets called recursively.
444 function tree{T<:Unsigned}(io::IO, bt::Vector{Vector{T}}, counts::Vector{Int}, lidict::Dict, level::Int, combine::Bool, cols::Integer, maxdepth)
445 if level > maxdepth
446 return
447 end
448 # Organize backtraces into groups that are identical up to this level
449 if combine
450 # Combine based on the line information
451 d = Dict{StackFrame,Vector{Int}}()
452 for i = 1:length(bt)
453 ip = bt[i][level+1]
454 key = lidict[ip]
455 indx = Base.ht_keyindex(d, key)
456 if indx == -1
457 d[key] = [i]
458 else
459 push!(d.vals[indx], i)
460 end
461 end
462 # Generate counts
463 dlen = length(d)
464 lilist = Array{StackFrame}(dlen)
465 group = Array{Vector{Int}}(dlen)
466 n = Array{Int}(dlen)
467 i = 1
468 for (key, v) in d
469 lilist[i] = key
470 group[i] = v
471 n[i] = sum(counts[v])
472 i += 1
473 end
474 else
475 # Combine based on the instruction pointer
476 d = Dict{T,Vector{Int}}()
477 for i = 1:length(bt)
478 key = bt[i][level+1]
479 indx = Base.ht_keyindex(d, key)
480 if indx == -1
481 d[key] = [i]
482 else
483 push!(d.vals[indx], i)
484 end
485 end
486 # Generate counts, and do the code lookup
487 dlen = length(d)
488 lilist = Array{StackFrame}(dlen)
489 group = Array{Vector{Int}}(dlen)
490 n = Array{Int}(dlen)
491 i = 1
492 for (key, v) in d
493 lilist[i] = lidict[key]
494 group[i] = v
495 n[i] = sum(counts[v])
496 i += 1
497 end
498 end
499 # Order the line information
500 if length(lilist) > 1
501 p = liperm(lilist)
502 lilist = lilist[p]
503 group = group[p]
504 n = n[p]
505 end
506 # Generate the string for each line
507 strs = tree_format(lilist, n, level, cols)
508 # Recurse to the next level
509 len = Int[length(x) for x in bt]
510 for i = 1:length(lilist)
511 if !isempty(strs[i])
512 println(io, strs[i])
513 end
514 idx = group[i]
515 keep = len[idx] .> level+1
516 if any(keep)
517 idx = idx[keep]
518 tree(io, bt[idx], counts[idx], lidict, level+1, combine, cols, maxdepth)
519 end
520 end
521 end

543 function callersf(matchfunc::Function, bt::Vector{UInt}, lidict)
544 counts = Dict{StackFrame, Int}()
545 lastmatched = false
546 for id in bt
547 if id == 0
548 lastmatched = false
549 continue
550 end
551 li = lidict[id]
552 if lastmatched
553 if haskey(counts, li)
554 counts[li] += 1
555 else
556 counts[li] = 1
557 end
558 end
559 lastmatched = matchfunc(li)
560 end
561 k = collect(keys(counts))
562 v = collect(values(counts))
563 p = sortperm(v, rev=true)
564 [(v[i], k[i]) for i in p]
565 end
