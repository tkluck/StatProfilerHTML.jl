205 dopushdisplay && pushdisplay(d)
206 repl_channel, response_channel = backend.repl_channel, backend.response_channel
207 hit_eof = false
208 while true
209 Base.reseteof(repl.terminal)
210 write(repl.terminal, "julia> ")
211 line = ""
212 ast = nothing
213 interrupted = false
214 while true
215 try
216 line *= readline(repl.terminal)
217 catch e
218 if isa(e,InterruptException)
219 try # raise the debugger if present
220 ccall(:jl_raise_debugger, Int, ())
221 end
222 line = ""
223 interrupted = true
224 break
225 elseif isa(e,EOFError)
226 hit_eof = true
227 break
228 else
229 rethrow()
230 end
231 end
232 ast = Base.parse_input_line(line)
233 (isa(ast,Expr) && ast.head == :incomplete) || break
234 end
235 if !isempty(line)
236 put!(repl_channel, (ast, 1))
237 val, bt = take!(response_channel)
238 if !ends_with_semicolon(line)
239 print_response(repl, val, bt, true, false)
240 end
241 end
242 write(repl.terminal, '\n')

355 function hist_from_file(hp, file)
356 hp.history_file = file
357 seek(file, 0)
358 countlines = 0
359 while true
360 mode = :julia
361 line = hist_getline(file)
362 isempty(line) && break
363 countlines += 1
364 line[1] != '#' &&
365 error(invalid_history_message, repr(line[1]), " at line ", countlines)
366 while !isempty(line)
367 m = match(r"^#\s*(\w+)\s*:\s*(.*?)\s*$", line)
368 m === nothing && break
369 if m.captures[1] == "mode"
370 mode = Symbol(m.captures[2])
371 end
372 line = hist_getline(file)
373 countlines += 1
374 end
375 isempty(line) && break
376 # Make sure starts with tab
377 line[1] == ' ' &&
378 error(munged_history_message, countlines)
379 line[1] != '\t' &&
380 error(invalid_history_message, repr(line[1]), " at line ", countlines)
381 lines = String[]
382 while !isempty(line)
383 push!(lines, chomp(line[2:end]))
384 eof(file) && break
385 ch = Char(Base.peek(file))
386 ch == ' ' && error(munged_history_message, countlines)
387 ch != '\t' && break
388 line = hist_getline(file)
389 countlines += 1
390 end
391 push!(hp.modes, mode)
392 push!(hp.history, join(lines, '\n'))
393 end
394 seekend(file)
395 hp.start_idx = length(hp.history)
396 hp
397 end

427 function history_move(s::Union{LineEdit.MIState,LineEdit.PrefixSearchState}, hist::REPLHistoryProvider, idx::Int, save_idx::Int = hist.cur_idx)
428 max_idx = length(hist.history) + 1
429 @assert 1 <= hist.cur_idx <= max_idx
430 (1 <= idx <= max_idx) || return :none
431 idx != hist.cur_idx || return :none
432
433 # save the current line
434 if save_idx == max_idx
435 hist.last_mode = LineEdit.mode(s)
436 hist.last_buffer = copy(LineEdit.buffer(s))
437 else
438 hist.history[save_idx] = LineEdit.input_string(s)
439 hist.modes[save_idx] = mode_idx(hist, LineEdit.mode(s))
440 end
441
442 # load the saved line
443 if idx == max_idx
444 last_buffer = hist.last_buffer
445 LineEdit.transition(s, hist.last_mode) do
446 LineEdit.replace_line(s, last_buffer)
447 end
448 hist.last_mode = nothing
449 hist.last_buffer = IOBuffer()
450 else
451 if haskey(hist.mode_mapping, hist.modes[idx])
452 LineEdit.transition(s, hist.mode_mapping[hist.modes[idx]]) do
453 LineEdit.replace_line(s, hist.history[idx])
454 end
455 else
456 return :skip
457 end
458 end
459 hist.cur_idx = idx
460
461 return :ok
462 end

557 function history_search(hist::REPLHistoryProvider, query_buffer::IOBuffer, response_buffer::IOBuffer,
558 backwards::Bool=false, skip_current::Bool=false)
559
560 qpos = position(query_buffer)
561 qpos > 0 || return true
562 searchdata = beforecursor(query_buffer)
563 response_str = String(response_buffer)
564
565 # Alright, first try to see if the current match still works
566 a = position(response_buffer) + 1
567 b = min(endof(response_str), prevind(response_str, a + sizeof(searchdata)))
568
569 !skip_current && searchdata == response_str[a:b] && return true
570
571 searchfunc, searchstart, skipfunc = backwards ? (rsearch, b, prevind) :
572 (search, a, nextind)
573 skip_current && (searchstart = skipfunc(response_str, searchstart))
574
575 # Start searching
576 # First the current response buffer
577 if 1 <= searchstart <= endof(response_str)
578 match = searchfunc(response_str, searchdata, searchstart)
579 if match != 0:-1
580 seek(response_buffer, prevind(response_str, first(match)))
581 return true
582 end
583 end
584
585 # Now search all the other buffers
586 idxs = backwards ? ((hist.cur_idx-1):-1:1) : ((hist.cur_idx+1):length(hist.history))
587 for idx in idxs
588 h = hist.history[idx]
589 match = searchfunc(h, searchdata)
590 if match != 0:-1 && h != response_str && haskey(hist.mode_mapping, hist.modes[idx])
591 truncate(response_buffer, 0)
592 write(response_buffer, h)
593 seek(response_buffer, prevind(h, first(match)))
594 hist.cur_idx = idx
595 return true
596 end
597 end
598
599 return false
600 end

695 function setup_interface(repl::LineEditREPL; hascolor = repl.hascolor, extra_repl_keymap = Dict{Any,Any}[])
696 ###
697 #
698 # This function returns the main interface that describes the REPL
699 # functionality, it is called internally by functions that setup a
700 # Terminal-based REPL frontend, but if you want to customize your REPL
701 # or embed the REPL in another interface, you may call this function
702 # directly and append it to your interface.
703 #
704 # Usage:
705 #
706 # repl_channel,response_channel = Channel(),Channel()
707 # start_repl_backend(repl_channel, response_channel)
708 # setup_interface(REPLDisplay(t),repl_channel,response_channel)
709 #
710 ###
711
712 ###
713 # We setup the interface in two stages.
714 # First, we set up all components (prompt,rsearch,shell,help)
715 # Second, we create keymaps with appropriate transitions between them
716 # and assign them to the components
717 #
718 ###
719
720 ############################### Stage I ################################
721
722 # This will provide completions for REPL and help mode
723 replc = REPLCompletionProvider(repl)
724
725 # Set up the main Julia prompt
726 julia_prompt = Prompt("julia> ";
727 # Copy colors from the prompt object
728 prompt_prefix = hascolor ? repl.prompt_color : "",
729 prompt_suffix = hascolor ?
730 (repl.envcolors ? Base.input_color : repl.input_color) : "",
731 keymap_func_data = repl,
732 complete = replc,
733 on_enter = return_callback)
734
735 # Setup help mode
736 help_mode = Prompt("help?> ",
737 prompt_prefix = hascolor ? repl.help_color : "",
738 prompt_suffix = hascolor ?
739 (repl.envcolors ? Base.input_color : repl.input_color) : "",
740 keymap_func_data = repl,
741 complete = replc,
742 # When we're done transform the entered line into a call to help("$line")
743 on_done = respond(Docs.helpmode, repl, julia_prompt))
744
745 # Set up shell mode
746 shell_mode = Prompt("shell> ";
747 prompt_prefix = hascolor ? repl.shell_color : "",
748 prompt_suffix = hascolor ?
749 (repl.envcolors ? Base.input_color : repl.input_color) : "",
750 keymap_func_data = repl,
751 complete = ShellCompletionProvider(repl),
752 # Transform "foo bar baz" into `foo bar baz` (shell quoting)
753 # and pass into Base.repl_cmd for processing (handles `ls` and `cd`
754 # special)
755 on_done = respond(repl, julia_prompt) do line
756 Expr(:call, :(Base.repl_cmd), macroexpand(Expr(:macrocall, Symbol("@cmd"),line)), outstream(repl))
757 end)
758
759
760 ################################# Stage II #############################
761
762 # Setup history
763 # We will have a unified history for all REPL modes
764 hp = REPLHistoryProvider(Dict{Symbol,Any}(:julia => julia_prompt,
765 :shell => shell_mode,
766 :help => help_mode))
767 if repl.history_file
768 try
769 f = open(find_hist_file(), true, true, true, false, false)
770 finalizer(replc, replc->close(f))
771 hist_from_file(hp, f)
772 catch e
773 print_response(repl, e, catch_backtrace(), true, Base.have_color)
774 println(outstream(repl))
775 info("Disabling history file for this session.")
776 repl.history_file = false
777 end
778 end

780 julia_prompt.hist = hp
781 shell_mode.hist = hp
782 help_mode.hist = hp
783
784 julia_prompt.on_done = respond(x->Base.parse_input_line(x,filename=repl_filename(repl,hp)), repl, julia_prompt)
785
786
787 search_prompt, skeymap = LineEdit.setup_search_keymap(hp)
788 search_prompt.complete = LatexCompletions()
789
790 # Canonicalize user keymap input
791 if isa(extra_repl_keymap, Dict)
792 extra_repl_keymap = [extra_repl_keymap]

795 const repl_keymap = AnyDict(
796 ';' => function (s,o...)
797 if isempty(s) || position(LineEdit.buffer(s)) == 0
798 buf = copy(LineEdit.buffer(s))
799 transition(s, shell_mode) do
800 LineEdit.state(s, shell_mode).input_buffer = buf
801 end
802 else
803 edit_insert(s, ';')
804 end
805 end,
806 '?' => function (s,o...)
807 if isempty(s) || position(LineEdit.buffer(s)) == 0
808 buf = copy(LineEdit.buffer(s))
809 transition(s, help_mode) do
810 LineEdit.state(s, help_mode).input_buffer = buf
811 end
812 else
813 edit_insert(s, '?')
814 end
815 end,
816
817 # Bracketed Paste Mode
818 "\e[200~" => (s,o...)->begin
819 input = LineEdit.bracketed_paste(s) # read directly from s until reaching the end-bracketed-paste marker
820 sbuffer = LineEdit.buffer(s)
821 curspos = position(sbuffer)
822 seek(sbuffer, 0)
823 shouldeval = (nb_available(sbuffer) == curspos && search(sbuffer, UInt8('\n')) == 0)
824 seek(sbuffer, curspos)
825 if curspos == 0
826 # if pasting at the beginning, strip leading whitespace
827 input = lstrip(input)
828 end
829 if !shouldeval
830 # when pasting in the middle of input, just paste in place
831 # don't try to execute all the WIP, since that's rather confusing
832 # and is often ill-defined how it should behave
833 edit_insert(s, input)
834 return
835 end
836 edit_insert(sbuffer, input)
837 input = takebuf_string(sbuffer)
838 oldpos = start(input)
839 firstline = true
840 while !done(input, oldpos) # loop until all lines have been executed
841 ast, pos = Base.syntax_deprecation_warnings(false) do
842 Base.parse(input, oldpos, raise=false)
843 end
844 if (isa(ast, Expr) && (ast.head == :error || ast.head == :continue || ast.head == :incomplete)) ||
845 (done(input, pos) && !endswith(input, '\n'))
846 # remaining text is incomplete (an error, or parser ran to the end but didn't stop with a newline):
847 # Insert all the remaining text as one line (might be empty)
848 tail = input[oldpos:end]
849 if !firstline
850 # strip leading whitespace, but only if it was the result of executing something
851 # (avoids modifying the user's current leading wip line)
852 tail = lstrip(tail)
853 end
854 LineEdit.replace_line(s, tail)
855 LineEdit.refresh_line(s)
856 break
857 end
858 # get the line and strip leading and trailing whitespace
859 line = strip(input[oldpos:prevind(input, pos)])
860 if !isempty(line)
861 # put the line on the screen and history
862 LineEdit.replace_line(s, line)
863 LineEdit.commit_line(s)
864 # execute the statement
865 terminal = LineEdit.terminal(s) # This is slightly ugly but ok for now
866 raw!(terminal, false) && disable_bracketed_paste(terminal)
867 LineEdit.mode(s).on_done(s, LineEdit.buffer(s), true)
868 raw!(terminal, true) && enable_bracketed_paste(terminal)
869 end
870 oldpos = pos
871 firstline = false
872 end
873 end,
874 )
875
876 prefix_prompt, prefix_keymap = LineEdit.setup_prefix_keymap(hp, julia_prompt)
877
878 a = Dict{Any,Any}[skeymap, repl_keymap, prefix_keymap, LineEdit.history_keymap, LineEdit.default_keymap, LineEdit.escape_defaults]
879 prepend!(a, extra_repl_keymap)
880
881 julia_prompt.keymap_dict = LineEdit.keymap(a)
882
883 mk = mode_keymap(julia_prompt)
884
885 b = Dict{Any,Any}[skeymap, mk, prefix_keymap, LineEdit.history_keymap, LineEdit.default_keymap, LineEdit.escape_defaults]
886 prepend!(b, extra_repl_keymap)
887
888 shell_mode.keymap_dict = help_mode.keymap_dict = LineEdit.keymap(b)
889
890 ModalInterface([julia_prompt, shell_mode, help_mode, search_prompt, prefix_prompt])
891 end

934 function ends_with_semicolon(line)
935 match = rsearch(line, ';')
936 if match != 0
937 for c in line[(match+1):end]
938 isspace(c) || return c == '#'
939 end
940 return true
941 end
942 return false
943 end

945 function run_frontend(repl::StreamREPL, backend::REPLBackendRef)
946 have_color = Base.have_color
947 banner(repl.stream, have_color)
948 d = REPLDisplay(repl)
949 dopushdisplay = !in(d,Base.Multimedia.displays)
950 dopushdisplay && pushdisplay(d)
951 repl_channel, response_channel = backend.repl_channel, backend.response_channel
952 while !eof(repl.stream)
953 if have_color
954 print(repl.stream,repl.prompt_color)
955 end
956 print(repl.stream, "julia> ")
957 if have_color
958 print(repl.stream, input_color(repl))
959 end
960 line = readline(repl.stream)
961 if !isempty(line)
962 ast = Base.parse_input_line(line)
963 if have_color
964 print(repl.stream, Base.color_normal)
965 end
966 put!(repl_channel, (ast, 1))
967 val, bt = take!(response_channel)
968 if !ends_with_semicolon(line)
969 print_response(repl, val, bt, true, have_color)
970 end
971 end
972 end
973 # Terminate Backend
974 put!(repl_channel, (nothing, -1))
975 dopushdisplay && popdisplay(d)
976 end
