39 function mapfoldl_impl(f, op, v0, itr, i)
40 # Unroll the while loop once; if v0 is known, the call to op may
41 # be evaluated at compile time
42 if done(itr, i)
43 return r_promote(op, v0)
44 else
45 (x, i) = next(itr, i)
46 v = op(r_promote(op, v0), f(x))
47 while !done(itr, i)
48 @inbounds (x, i) = next(itr, i)
49 v = op(v, f(x))
50 end
51 return v
52 end
53 end

72 function mapfoldr_impl(f, op, v0, itr, i::Integer)
73 # Unroll the while loop once; if v0 is known, the call to op may
74 # be evaluated at compile time
75 if i == 0
76 return r_promote(op, v0)
77 else
78 x = itr[i]
79 v = op(f(x), r_promote(op, v0))
80 while i > 1
81 x = itr[i -= 1]
82 v = op(f(x), v)
83 end
84 return v
85 end
86 end

96 function mapreduce_impl(f, op, A::AbstractArray, ifirst::Integer, ilast::Integer, blksize::Int=pairwise_blocksize(f, op))
97 if ifirst + blksize > ilast
98 # sequential portion
99 fx1 = r_promote(op, f(A[ifirst]))
100 fx2 = r_promote(op, f(A[ifirst + 1]))
101 v = op(fx1, fx2)
102 @simd for i = ifirst + 2 : ilast
103 @inbounds Ai = A[i]
104 v = op(v, f(Ai))
105 end
106 return v
107 else
108 # pairwise portion
109 imid = (ifirst + ilast) >> 1
110 v1 = mapreduce_impl(f, op, A, ifirst, imid, blksize)
111 v2 = mapreduce_impl(f, op, A, imid+1, ilast, blksize)
112 return op(v1, v2)
113 end
114 end

149 function _mapreduce{T}(f, op, ::LinearFast, A::AbstractArray{T})
150 inds = linearindices(A)
151 n = length(inds)
152 @inbounds begin
153 if n == 0
154 return mr_empty(f, op, T)
155 elseif n == 1
156 return r_promote(op, f(A[inds[1]]))
157 elseif n < 16
158 fx1 = r_promote(op, f(A[inds[1]]))
159 fx2 = r_promote(op, f(A[inds[2]]))
160 s = op(fx1, fx2)
161 i = inds[2]
162 while i < last(inds)
163 Ai = A[i+=1]
164 s = op(s, f(Ai))
165 end
166 return s
167 else
168 return mapreduce_impl(f, op, A, first(inds), last(inds))
169 end
170 end
171 end

240 # Kahan (compensated) summation: O(1) error growth, at the expense
241 # of a considerable increase in computational expense.
242 function sum_kbn{T<:AbstractFloat}(A::AbstractArray{T})
243 c = r_promote(+, zero(T)::T)
244 if isempty(A)
245 return c
246 end
247 inds = linearindices(A)
248 s = A[first(inds)] + c
249 for i in first(inds)+1:last(inds)
250 @inbounds Ai = A[i]
251 t = s + Ai
252 if abs(s) >= abs(Ai)
253 c += ((s-t) + Ai)
254 else
255 c += ((Ai-t) + s)
256 end
257 s = t
258 end
259 s + c
260 end

354 """
355 extrema(itr) -> Tuple
356
357 Compute both the minimum and maximum element in a single pass, and return them as a 2-tuple.
358
359 ```jldoctest
360 julia> extrema(2:10)
361 (2,10)
362
363 julia> extrema([9,pi,4.5])
364 (3.141592653589793,9.0)
365 ```
366 """
367 function extrema(itr)
368 s = start(itr)
369 done(itr, s) && throw(ArgumentError("collection must be non-empty"))
370 (v, s) = next(itr, s)
371 while v != v && !done(itr, s)
372 (x, s) = next(itr, s)
373 v = x
374 end
375 vmin = v
376 vmax = v
377 while !done(itr, s)
378 (x, s) = next(itr, s)
379 if x > vmax
380 vmax = x
381 elseif x < vmin
382 vmin = x
383 end
384 end
385 return (vmin, vmax)
386 end
